启动：
cd
mkdir ~/pgsql
cd postgres
make clean
./configure --prefix=/home/hututuzi/pgsql/ --enable-debug --enable-depend --enable-cassert CFLAGS=-O0
make
make check
make install
 (如果有pgdata了就需要执行这个)
rm -rf ~/pgdata
mkdir ~/pgdata
cd
pgsql/bin/initdb -D pgdata
pgsql/bin/pg_ctl -D pgdata start
pgsql/bin/psql -h localhost -d postgres
